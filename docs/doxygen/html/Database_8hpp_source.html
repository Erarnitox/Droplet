49template <typename... Types>
50[[nodiscard("You need to check if the Query was executed on the Database!")]] bool execQuery(const std::string& query,
51 Types&&... args) noexcept {
52 static int times = 0;
53 try {
54 NativeDatabase::Transaction txn(*Database::getConnection());
55
56 // perform the database transaction
57 const NativeDatabase::ParameterList params{std::forward<Types>(args)...};
58 const NativeDatabase::Result result = txn.exec(query, params);
59
60 txn.commit();
61
62 times = 0;
63 return true;
64 } catch (const NativeDatabase::BrokenConnectionException& e) {
65 ++times;
66 if (times > 10) {
67 times = 0;
68 return false;
69 }
70 Database::reconnect();
71 return execQuery(query, args...);
72 } catch (const std::exception& e) {
73 std::cerr << "Error: " << e.what() << "\nInvalid exec for query:\n" << query << std::endl;
74 return false;
75 }
76}

120template <typename... Types>
121[[nodiscard]] std::vector<RowDTOAdapter> execSelectAll(const std::string& query, Types&&... args) noexcept {
122 static int times = 0;
123 try {
124 NativeDatabase::Transaction txn(*Database::getConnection());
125
126 // perform the database transaction
127 const NativeDatabase::ParameterList params{std::forward<Types>(args)...};
128 const NativeDatabase::Result result = txn.exec(query, params);
129
130 txn.commit();
131
132 times = 0;
133
134 std::vector<RowDTOAdapter> selection;
135 for (int i{0}; i < result.size(); ++i) {
136 selection.emplace_back(result[i]);
137 }
138
139 return selection;
140 } catch (const NativeDatabase::BrokenConnectionException& e) {
141 ++times;
142 if (times > 10) {
143 times = 0;
144 return {NativeDatabase::Row()};
145 }
146 Database::reconnect();
147 return execSelectAll(query, args...);
148 } catch (const std::exception& e) {
149 std::cerr << "Error: " << e.what() << "\nInvalid selection for query:\n" << query << std::endl;
150 return {{}}; // return an empty selection
151 }
