31static inline auto handle_global_slash_commands(const dpp::slashcommand_t &event, dpp::cluster &bot) noexcept -> void {
32 if (event.command.get_command_name() == "reaction_role") {
33 if (!Core::isAdmin(event.command.member)) {
34 event.reply(dpp::message("Only admins are allowed to use this command!").set_flags(dpp::m_ephemeral));
35 return;
36 }
37
38 const auto message_link{Core::getParameter(bot, event, "message_link")};
39 if (message_link.empty())
40 return;
41
42 const auto emoji{Core::getParameter(bot, event, "emoji")};
43 if (emoji.empty())
44 return;
45
46 const auto role{Core::getParameter(bot, event, "role")};
47 if (role.empty())
48 return;
49
50 const auto role_id{Core::getRoleId(role)};
51 if (role_id.empty()) {
52 event.reply(dpp::message("No valid Role was provided!").set_flags(dpp::m_ephemeral));
53 return;
54 }
55
56 const auto usable_emoji{emoji.starts_with("<:") ? emoji.substr(2, emoji.size() - 3) : emoji};
57 if (usable_emoji.empty()) {
58 event.reply(dpp::message("No valid emoji was provided!").set_flags(dpp::m_ephemeral));
59 return;
60 }
61
62 // indecies of slashes in the link
63 std::vector<size_t> slashes;
64
65 if (message_link.size() < 4 || !message_link.starts_with("http")) {
66 event.reply(dpp::message("The provided link is not a valid message link!").set_flags(dpp::m_ephemeral));
67 return;
68 }
69
70 // find the position of all slashes
71 for (size_t i{0};;) {
72 i = message_link.find("/", i + 1);
73 if (i == std::string::npos)
74 break;
75 slashes.push_back(i);
76 }
77
78 if (slashes.size() < 3) {
79 event.reply("The provided link is not a valid message link!");
80 return;
81 }
82
83 //--------------------------------------------------
84 // get message id from the link
85 size_t end_pos{message_link.size()};
86 size_t start_pos{slashes.back() + 1};
87 const auto &message_id{message_link.substr(start_pos, end_pos)};
88 //--------------------------------------------------
89
90 //--------------------------------------------------
91 // get message id from the link
92 slashes.pop_back();
93 end_pos = (start_pos - 2) - slashes.back();
94 start_pos = slashes.back() + 1;
95 const auto &channel_id{message_link.substr(start_pos, end_pos)};
96 //--------------------------------------------------
97
98 // Insert Reaction Role into Database
99 Database::insert_reaction_role_data(role_id, event.command.guild_id, message_id, usable_emoji);
100
101 // Let the bot react to the message
102 bot.message_add_reaction(message_id, channel_id, usable_emoji);
103
104 // send a confirmation to the admin
105 event.reply(
106 dpp::message(
107 fmt::format("Reaction Role Created!\nMessage: {}\nReaction: {}\nRole: {}", message_link, emoji, role))
108 .set_flags(dpp::m_ephemeral));
109 }
110}
